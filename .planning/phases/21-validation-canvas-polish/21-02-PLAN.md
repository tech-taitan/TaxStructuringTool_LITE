```typescript
const activeJurisdictions = useMemo(() => {
  const jurisdictions = new Set<string>();
  for (const node of nodes) {
    if (node.data.jurisdiction) {
      jurisdictions.add(node.data.jurisdiction);
    }
  }
  return Array.from(jurisdictions).sort();
}, [nodes]);
```
